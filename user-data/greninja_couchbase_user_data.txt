#!/bin/bash
apt-get update
#Set env var permanently
echo "export ENV_NAME=greninja" >> /home/ubuntu/.profile
#Set env var for this session
export ENV_NAME=greninja
#Mount /data drive (assumes 30G size drive)
drive_name=$(lsblk | grep 30G | cut -c 1-5)
mount /dev/"${drive_name}" /data
chown -R couchbase:couchbase /data/
